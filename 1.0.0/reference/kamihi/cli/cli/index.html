def main(
    ctx: typer.Context,
    config: Annotated[
        Path | None,
        typer.Option(
            ...,
            help="Path to the Kamihi configuration file",
            exists=True,
            file_okay=True,
            dir_okay=False,
            readable=True,
            show_default="kamihi.yaml",
        ),
    ] = None,
) -> None:
    """
    Kamihi CLI utility.

    This utility provides commands to manage and interact with the Kamihi framework.
    """
    ctx.obj = Context()
    if config is not None:
        ctx.obj.config = config
